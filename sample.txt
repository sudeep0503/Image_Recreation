from PIL import Image
import numpy as np

def load_channel(filename):
    rows = []

    with open(filename, "r") as f:
        for line in f:

            # Skip markdown lines
            if line.startswith("#"):
                continue

            if line.startswith("```"):
                continue

            line = line.strip()

            if not line:
                continue

            rows.append(list(map(int, line.split())))

    return np.array(rows, dtype=np.uint8)


# Load channels
R = load_channel("R.txt")
G = load_channel("G.txt")
B = load_channel("B.txt")

print("R shape:", R.shape)
print("G shape:", G.shape)
print("B shape:", B.shape)

# Combine channels
img_array = np.stack((R, G, B), axis=2)

print("Image shape:", img_array.shape)

# Create image
img = Image.fromarray(img_array)

# Save image
img.save("recreated.png")

print("Image recreated successfully!")
